# read in the data as text
df = pd.read_csv('IBM_adjusted.txt',header=None)
df.columns = ['date','time','open','high','low','close','volume']

# combine the date/time as a string, convert to datetime64[ns]
df['date'] = pd.to_datetime(df['date'] + ' ' + df['time'],infer_datetime_format=True)
df.pop('time')

# set the date as the index
df = df.set_index('date')

# write to hdf5
df.to_hdf('ibm.h5','df',mode='w')
