def get_data():
    import urllib as web
    date = "2014-10-30"

    stock='IBM.N'
    baseUrl='http://netfonds.no/quotes/tradedump.php?csv_format=csv'
    fullUrl=baseUrl+'&date='+date.replace("-","")+'&paper='+stock

    webBuffer=web.urlopen(fullUrl)
    webData=pd.read_csv(webBuffer)
    webData.to_csv('ibm.csv')
